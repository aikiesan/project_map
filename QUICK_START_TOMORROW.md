def generate_abnt_citations() -> str:
    """Generate ABNT format citations"""
    db = get_reference_database()
    citations = []

    for ref in db.references.values():
        if ref.citation_abnt:
            citations.append(ref.citation_abnt)

    return "\n\n".join(citations)
